function doGet() { try { // --- 【SaaS設定エリア】 --- const START_DATE = new Date('2026-04-29T00:00:00+09:00'); const VALID_DAYS = 5; // -------------------------- const now = new Date(); const diffTime = now.getTime() - START_DATE.getTime(); let elapsedDays = Math.floor(diffTime / (1000 * 60 * 60 * 24)); if (elapsedDays < 0) elapsedDays = 0; // 【1】時限装置 if (elapsedDays >= VALID_DAYS) { return createJsonResponse({ error: "一定の期間が経過したため、アクセスできません。" }); } // 【2】日替わり計算（2行目からスタート） const targetRow = 2 + elapsedDays; const ss = SpreadsheetApp.getActiveSpreadsheet(); // 読み込むシート名の配列 const sheetNames = ["心臓血管外科クイズ(易)", "心臓血管外科クイズ(普通)", "心臓血管外科クイズ(難)"]; const quizResults = []; // 各シートを順番に処理 for (let i = 0; i < sheetNames.length; i++) { const difficulty = sheetNames[i]; const sheet = ss.getSheetByName(difficulty); // シートが存在しない場合のエラーハンドリング if (!sheet) { quizResults.push({ difficulty: difficulty, error: `${difficulty}シートが見つかりません。` }); continue; } const lastRow = sheet.getLastRow(); // その難易度の問題が尽きている場合の処理 if (targetRow > lastRow) { quizResults.push({ difficulty: difficulty, isPlaceholder: true, question: `現在「${difficulty}」の新しい問題を準備中です。`, image: "", option1: "", option2: "", option3: "", option4: "", hint: "", answer: "", explanation: "追加をお待ちください。" }); continue; } // 【高速化】1行分（A列〜L列の12列）のデータを一括取得 const rowData = sheet.getRange(targetRow, 1, 1, 12).getValues()[0]; // スプレッドシートの列定義（0始まりのインデックス） // A:0(ID), B:1(出典), C:2(難易度), D:3(問題文), E:4(画像), F:5(選1), G:6(選2), H:7(選3), I:8(選4), J:9(ヒント), K:10(回答), L:11(解説) const questionData = { difficulty: difficulty, question: String(rowData[3] || "").trim(), image: String(rowData[4] || "").trim(), option1: String(rowData[5] || "").trim(), option2: String(rowData[6] || "").trim(), option3: String(rowData[7] || "").trim(), option4: String(rowData[8] || "").trim(), hint: String(rowData[9] || "").trim(), answer: String(rowData[10] || "").trim(), explanation: String(rowData[11] || "").trim(), dayCount: elapsedDays + 1 }; quizResults.push(questionData); } // 3問分の配列データをJSONとして返す return createJsonResponse(quizResults); } catch (error) { return createJsonResponse({ error: "システムエラー: " + error.message }); } } function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data, null, 2)) .setMimeType(ContentService.MimeType.JSON); }
